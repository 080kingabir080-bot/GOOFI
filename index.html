let games = JSON.parse(localStorage.getItem('pp_ultra_v1')) || []; let users = JSON.parse(localStorage.getItem('pp_users_v1')) || []; let currentUser = JSON.parse(sessionStorage.getItem('pp_active')) || null; function nav(id) { document.querySelectorAll('.page').forEach(p => p.classList.remove('active')); document.getElementById(id).classList.add('active'); if(id === 'homePage') renderGames(); if(id === 'adminPage') { renderAdminGames(); renderAdminTools(); // קריאה חדשה להצגת הכלים בניהול } } function openAdminTool(id) { document.getElementById(id).style.display = 'flex'; } function closeAdminTool(id) { document.getElementById(id).style.display = 'none'; } function processRegister() { const u = document.getElementById('r_user').value; const p = document.getElementById('r_pass').value; if(!u || !p) return; users.push({name: u, pass: p, isPrem: false}); localStorage.setItem('pp_users_v1', JSON.stringify(users)); alert("החשבון נוצר!"); nav('loginPage'); } function processLogin() { const u = document.getElementById('l_user').value; const p = document.getElementById('l_pass').value; if(u === 'admin' && p === 'admin') return nav('adminPage'); const found = users.find(x => x.name === u && x.pass === p); if(found) { currentUser = found; sessionStorage.setItem('pp_active', JSON.stringify(found)); checkPremium(); nav('homePage'); } else { alert("טעות בפרטים"); } } function tryOpenGame(id) { const g = games.find(x => x.id == id); if(!g) return; document.getElementById('ovTitle').innerText = g.title; document.getElementById('gameLoader').style.display = 'flex'; document.getElementById('gameFrame').classList.add('hidden'); const blob = new Blob([g.code], {type: 'text/html'}); document.getElementById('gameFrame').src = URL.createObjectURL(blob); document.getElementById('gameOverlay').style.display = 'flex'; } function closeGame() { document.getElementById('gameOverlay').style.display = 'none'; document.getElementById('gameFrame').src = ''; } function saveNewItem(type) { let n, c, icon; if(type === 'game') { n = document.getElementById('adm_name').value; c = document.getElementById('adm_code').value; if(!n || !c) return; games.push({id: Date.now(), title: n, code: c, type: 'game'}); closeAdminTool('addGameTool'); } else { n = document.getElementById('tool_name').value; icon = document.getElementById('tool_emoji').value || '🛠️'; c = document.getElementById('tool_code').value; if(!n || !c) return; games.push({id: Date.now(), title: n, code: c, type: 'tool', emoji: icon}); closeAdminTool('addToolWindow'); } localStorage.setItem('pp_ultra_v1', JSON.stringify(games)); if(type === 'game') renderAdminGames(); else renderAdminTools(); alert("התוכן נשמר בהצלחה!"); } // מציג את הכלים בתוך פאנל הניהול ככפתורים שמישים function renderAdminTools() { const adminPanel = document.querySelector('#adminPage .flex-1'); // מנקים את הכלים הישנים שהוספו דינמית (משאירים את כפתורי המערכת) const dynamicTools = adminPanel.querySelectorAll('.dynamic-tool'); dynamicTools.forEach(t => t.remove()); const toolsOnly = games.filter(g => g.type === 'tool'); toolsOnly.forEach(tool => { const btn = document.createElement('div'); btn.className = 'admin-icon dynamic-tool'; btn.onclick = () => tryOpenGame(tool.id); btn.innerHTML = ` ${tool.emoji} ${tool.title} `; adminPanel.appendChild(btn); }); } function renderAdminGames() { document.getElementById('adminGameList').innerHTML = games.map(g => `
${g.type === 'tool' ? '🛠️' : '🎮'} ${g.title}
`).join(''); } function delGame(id) { if(confirm("למחוק?")) { games = games.filter(g => g.id != id); localStorage.setItem('pp_ultra_v1', JSON.stringify(games)); renderAdminGames(); renderAdminTools(); } } function checkPremium() { if(currentUser?.isPrem) { document.body.classList.add('is-premium'); document.getElementById('logoAccent').style.color = 'var(--premium)'; } } function renderGames() { const grid = document.getElementById('gameGrid'); // כאן השינוי החשוב: מסננים רק פריטים שהם 'game' const gamesOnly = games.filter(g => g.type === 'game'); grid.innerHTML = gamesOnly.map(g => `
${currentUser?.isPrem ? '💎' : '⚡'}
${g.title}
PLAY
`).join(''); const auth = document.getElementById('authArea'); if(currentUser) { auth.innerHTML = ` ${!currentUser.isPrem ? '' : ''}
${currentUser.name}
`; } } function startPayment() { document.getElementById('buyButtonWrapper').classList.add('hidden'); document.getElementById('verificationUI').classList.remove('hidden'); setTimeout(() => { document.getElementById('pBar').style.width = "100%"; }, 100); setTimeout(() => { document.getElementById('verificationUI').classList.add('hidden'); document.getElementById('successUI').classList.remove('hidden'); }, 6100); } function applyPremium() { currentUser.isPrem = true; const idx = users.findIndex(x => x.name === currentUser.name); users[idx].isPrem = true; localStorage.setItem('pp_users_v1', JSON.stringify(users)); sessionStorage.setItem('pp_active', JSON.stringify(currentUser)); location.reload(); } window.onload = () => { checkPremium(); renderGames(); if(currentUser?.name === 'admin') nav('adminPage'); else if(localStorage.getItem('pp_active')) nav('homePage'); // תיקון קטן לטעינה };
